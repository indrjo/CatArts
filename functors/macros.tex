% Other shortcuts
\newcommand\naturaltr[5]{
  \begin{tikzcd}[ampersand replacement=\&]
    #2
      \ar["#4"{name=A}, r, bend left=35]
      \ar["#5"{name=B}, r, bend right=35, swap]
    \& #3
    \ar["#1", from=A, to=B, natural] 
  \end{tikzcd}
}

}
